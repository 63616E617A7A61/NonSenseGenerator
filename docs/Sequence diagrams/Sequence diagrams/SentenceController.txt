@startuml

participant "SentenceController" as sc
participant "InptutSentence" as is
participant "OutputSentence" as os
participant "ApiController" as ac


group SentenceController()
sc <- is : sentence

is -> os : provides sentence
sc <- os : generated sentence
end

group getSyntaxTree()
sc -> is : getSyntaxTree()

is -> ac : getSyntaxTree(value)

activate ac
is <- ac : SyntaxTree
deactivate ac

sc <- is : SyntaxTree
end

group getToxicity()
sc -> os : getToxicity()
sc <- os : toxicity
end
@enduml
